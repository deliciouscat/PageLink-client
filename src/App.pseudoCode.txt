# 개요
`components`의 내용들 중 어떤 것을 최종적으로 표시할지 결정함.

# Pseudo-code
from components import app-header, bookmark-page, explore-page

if app-header.AppHeader.currentMode == "bookmark":
    main_display = bookmark-page
elif app-header.AppHeader.currentMode == "explore":
    main_display = explore-page
else:
    Box(
      text: {content: "Error", alignment: "center"}
    )

VerticalGrid(
    elements: [
        app-header.AppHeader,
        main_display
    ]
)
