# 기능 개요
`CollectionItem.vue` 컴포넌트를 나열하는 컴포넌트. `AppHeader.vue`와 `ToolBar.vue`와 완전히 연동하여 검색/Collection 생성 기능이 작동한다.

# Pseudo-code
import app-header/AppHeader.vue
import CollectionItem.vue
import ChromeAPI

function keyword_filtering(keyword: str, collections: list) :
    return [c for c in collections if c.contains(keyword)!=True]

if AppHeader.ToolBar.search_button.activated:
    collections2display = keyword_filtering(AppHeader.ToolBar.input_text, ChromeAPI.bookmarks.directories.listup)
else:
    collections2display = ChromeAPI.bookmarks.directories.listup

VerticalGrid(
    elements: [CollectionItem(c) for c in collections2display]
)

if (AppHeader.ToolBar.add_button.activated) and (key_pressed.enter):
    if AppHeader.ToolBar.input_text in ChromeAPI.bookmarks.directories.listup:
        console.log("Collection already exists")
    else :
        ChromeAPI.bookmarks.directories.add(AppHeader.ToolBar.input_text)
